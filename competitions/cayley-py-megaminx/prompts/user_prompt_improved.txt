def _inverse_name(move: str) -> str:

    return [a[j] for j in b]

def _emit_face(face: str, exp: int, face_order: dict[str, int]) -> list[tuple[str, int]]:
    order = face_order[face]
    value = exp % order
    if value == 0:
        return []
    if value <= order // 2:
        return [(face, value)]
    return [(face, value - order)]


def _emit_tokens(block: tuple[str, int]) -> list[str]:
    face, exp = block
    return [face] * exp if exp > 0 else ['-' + face] * (-exp)


def _reduce_commuting_word(moves: Sequence[str], generators: Dict[str, List[int]]) -> list[str]:

    changed = True
    while changed:
        changed = False
        merged: list[tuple[str, int]] = []
        i = 0
        while i < len(blocks):
            face, exp = blocks[i]
            j = i + 1
            while j < len(blocks) and blocks[j][0] == face:
                exp += blocks[j][1]
                j += 1
            norm = _emit_face(face, exp, face_order)
            if norm != blocks[i:j]:
                changed = True
            merged.extend(norm)
            i = j
        blocks = merged
        i = 0
        while i + 1 < len(blocks):

                changed = True
                if i:

    out: list[str] = []
    for block in blocks:

    return out


def _reverse_invert(moves: Sequence[str], inverse: dict[str, str]) -> list[str]:

def _decode_word(word: bytes, id_to_move: list[str]) -> list[str]:

    current = list(moves)
    stagnant_rounds = 0
    for _ in range(8):
        best = current

        left = _reduce_commuting_word(current, generators)
        left = _optimize_local_windows(left, generators, max_window=10)
        left = _reduce_commuting_word(left, generators)
        if len(left) < len(best):
            best = left

        right = _reverse_invert(current, inverse)
        right = _reduce_commuting_word(right, generators)
        right = _optimize_local_windows(right, generators, max_window=10)
        right = _reduce_commuting_word(right, generators)
        right = _reverse_invert(right, inverse)
        right = _reduce_commuting_word(right, generators)
        if len(right) < len(best):
            best = right

        if len(best) < len(current):
            current = best
            stagnant_rounds = 0
        else:
            stagnant_rounds += 1
            current = best
            if stagnant_rounds >= 2:
                break
    return current

def _apply_perm(state: List[int], perm: List[int]) -> List[int]:
    return [state[j] for j in perm]


def _apply_moves(vec: Sequence[int], moves: Sequence[str], generators: Dict[str, List[int]]) -> List[int]:
    state = list(vec)
    for move in moves:
        perm = generators.get(move)
        if perm is None:
            raise KeyError(move)
        state = _apply_perm(state, perm)
    return state


def solve(vec: Sequence[int]) -> Tuple[MoveOut, List[int]]:
    central, generators, lookup = _load_bundle()
    state = list(vec)
    if state == central:
        return [], list(state)
    state_key = ','.join(str(int(x)) for x in state)
    path = lookup.get(state_key)
    if path is not None:
        moves = [] if not path else [move for move in path.split('.') if move]
        return moves, _apply_moves(state, moves, generators)
    return 'UNSOLVED', list(state)
